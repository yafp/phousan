001package de.yafp.phousan;
002
003import android.app.backup.BackupAgentHelper;
004import android.app.backup.SharedPreferencesBackupHelper;
005
006/**
007 * Backup Agent - to backup the SharedPreference values
008 */
009public class BackupAgent extends BackupAgentHelper {
010    // The name of the SharedPreferences file
011    private static final String PREFS_NAME = "phousan_settings";
012
013    // A key to uniquely identify the set of backup data
014    private static final String PREFS_BACKUP_KEY = "prefs";
015
016    // Allocate a helper and add it to the backup agent
017    @Override
018    public void onCreate() {
019        SharedPreferencesBackupHelper helper = new SharedPreferencesBackupHelper(this, PREFS_NAME);
020        addHelper(PREFS_BACKUP_KEY, helper);
021    }
022}
