099    @Override
100    public boolean onOptionsItemSelected(MenuItem item) {
101        Log.d(TAG, "F: onOptionsItemSelected");
102
103        // log to Firebase
104        logFireBaseEvent("p___onOptionsItemSelected");
105
106        switch (item.getItemId()) {
107
108            // Menu: About
109            case R.id.about:
110                try {
111                    openAbout();
112                } catch (PackageManager.NameNotFoundException e) {
113                    e.printStackTrace();
114                }
115                break;
116
117            // Menu: Issues
118            case R.id.issues:
119                // log to Firebase
120                logFireBaseEvent("p___issues");
121
122                Uri uri_issues = Uri.parse("https://github.com/yafp/phousan/issues"); // missing 'http://' will cause crashed

124                startActivity(intent);

127            // Menu: GooglePlay
128            case R.id.googleplay:
129                // log to Firebase
130                logFireBaseEvent("p___googlePlay");
131

133                Intent intent3 = new Intent(Intent.ACTION_VIEW, uri_googlePlay);
134                startActivity(intent3);
135                return true;
136
137            // Menu: show history
138            case R.id.history_show:
139                showUsageDataPerDay();
140                return true;
141
142            // Menu: export history
143            case R.id.history_export:
144                exportUsageData();
145                return true;
146
147            // Menu: delete usage data
148            case R.id.usage_data_delete:
149                DialogInterface.OnClickListener dialogClickListener = new DialogInterface.OnClickListener() {
150                    @Override
151                    public void onClick(DialogInterface dialog, int which) {
152                        switch (which){
153                            case DialogInterface.BUTTON_POSITIVE:
154                                deleteAllSharedPreferences();
155                                break;
156
157                            default:
158                        }
159                    }
160                };
161
162                android.app.AlertDialog.Builder builder = new android.app.AlertDialog.Builder(this);
163                builder.setTitle(getResources().getString(R.string.really_delete_all_prefs_dialog_title));
164                builder.setMessage(getResources().getString(R.string.really_delete_all_prefs_dialog_text));
165                builder.setPositiveButton(getResources().getString(R.string.really_delete_all_prefs_dialog_yes), dialogClickListener);
166                builder.setNegativeButton(getResources().getString(R.string.really_delete_all_prefs_dialog__no), dialogClickListener);

170
171            // Menu: Recommend
172            case R.id.recommend:
173                recommendApp();
174                return true;
175
176            // Menu: service start
177            case R.id.service_start:
178                startService(new Intent(this, BackgroundService.class));
179                Log.d(TAG, "onClick - User wants to start the service");
180
181                // log to Firebase
182                logFireBaseEvent("p___serviceStart");
183
184                return true;
185
186            // Menu: service stop
187            case R.id.service_stop:
188                stopService(new Intent(this, BackgroundService.class));
189                Log.d(TAG, "onClick - User wants to stop the service");
190
191                // log to Firebase
192                logFireBaseEvent("p___serviceStop");
193                return true;
194
195            default:
196                break;
197        }
198        return true;
199    }

309    /**
310     * User wants to export usage data
311     * Generates usage data text which gets forwarded to an app of choice
312     */
313    private void exportUsageData() {
314        Log.d(TAG, "F: exportUsageData");
315
316        StringBuilder outputUsageHistory;
317        outputUsageHistory = new StringBuilder();
318        ArrayList<String> al= new ArrayList<>(); // stores the relevant prefs for later sorting
319
320        SharedPreferences settings = getSharedPreferences(PREFS_NAME, MODE_PRIVATE);
321
322        int record_amount; // counter
323
324        record_amount = 0;
325
326        // fetch all usage-data preference strings
327        for (String key : settings.getAll().keySet()) {
328            if(key.startsWith("20")) { // collect only usage-history data
329                Log.i(String.format("Shared Preference : %s - %s", PREFS_NAME, key), settings.getString(key, "error!"));
330
331                // add to array
332                al.add(key+";"+settings.getString(key, "error!")+";\n");
333
334                record_amount = record_amount +1; // counter
335            }
336            else {
337                Log.d(TAG, "Ignoring key, as it is not part of the daily usage history");
338            }
339        }
340
341        // calc average
342        if(record_amount == 0)
343        {
344            outputUsageHistory.append(getResources().getString(R.string.show_usage_history_per_day_empty_history));
345        }
346
347        // sort & reverse array-data
348        Collections.sort(al); // sort
349        Collections.reverse(al); //
350
351        // build output string
352        for (int i = 0; i < al.size(); i++) {
353            outputUsageHistory.append(al.get(i));
354        }
355
356        // forward information to external app
357        //
358        Intent share = new Intent(Intent.ACTION_SEND);
359        share.setType("text/plain");
360        share.addFlags(Intent.FLAG_ACTIVITY_NEW_DOCUMENT);
361
362        // Add data to the intent, the receiving app will decide what to do with it.
363        share.putExtra(Intent.EXTRA_SUBJECT, R.string.usage_data_export_title);
364        share.putExtra(Intent.EXTRA_TEXT, outputUsageHistory.toString());
365
366        startActivity(Intent.createChooser(share, getResources().getString(R.string.usage_data_export_title)));
367
368        // log to Firebase
369        logFireBaseEvent("p___exportUsageData");
370    }

651    /**
652     * fetches all usage data per day,
653     * calculates the average value
654     * and returns that value as a string
655     *
656     * @return the average usage count per day (overall)
657     */
658    private String getUsageDataAverage(){
659        Log.d(TAG, "F: getUsageDataAverage");
660
661        String averageUsageDataPerDay;
662
663        SharedPreferences settings = getSharedPreferences(PREFS_NAME, MODE_PRIVATE);
664
665        int record_amount; // counter
666        int record_sum; // usage sum
667        int record_avg; // calculated usage average
668
669        record_amount = 0;
670        record_sum = 0;
671
672        // print all settings
673        for (String key : settings.getAll().keySet()) {
674            if(key.startsWith("20")) { // collect only usage-history data
675                Log.i(String.format("Shared Preference : %s - %s", PREFS_NAME, key), settings.getString(key, "error!"));
676
677                record_amount = record_amount +1; // counter
678                record_sum = record_sum + Integer.parseInt(settings.getString(key, "error!"));
679            }
680            else {
681                Log.d(TAG, "Ignoring key, as it is not part of the daily usage history");
682            }
683        }
684
685        // calc average
686        if(record_amount == 0) {
687            averageUsageDataPerDay = "0";
688        }
689        else {
690            record_avg = record_sum / record_amount;
691            averageUsageDataPerDay = Integer.toString(record_avg);
692            Log.d(TAG, "Usage History Average: "+Integer.toString(record_avg));
693        }
694
695        // log to Firebase
696        logFireBaseEvent("p___getUsageDataAverage");
697
698        return averageUsageDataPerDay;
699    }
